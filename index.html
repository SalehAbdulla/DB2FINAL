Unit 10: PL/SQL Functions

Key Concepts

A function is a PL/SQL module that returns a value. Functions are reusable and compiled once, stored in your schema.

Function Structure

FUNCTION [schema.]name [(parameters)]
RETURN return_datatype
IS|AS
  [declarations]
BEGIN
  -- PL/SQL statements
  RETURN value;
[EXCEPTION
  exception_handlers]
END [name];

Important Points

  • Must have a RETURN statement in the executable section
  • Parameters are IN mode only (read-only)
  • Can be called from SELECT statements
  • Formal parameters = placeholders in header
  • Actual parameters = values passed during call
  • Drop with: DROP FUNCTION function_name;

Example: Calculate Age Function

CREATE OR REPLACE FUNCTION Calculate_Age (Birth_Date IN Date) 
RETURN REAL
IS
  Age REAL;
BEGIN
  Age := (SYSDATE - Birth_Date) / 365.25;
  RETURN Age;
EXCEPTION
  WHEN OTHERS THEN 
    RETURN NULL;
END Calculate_Age;

Unit 11: PL/SQL Procedures

Key Concepts

A procedure is a module that performs actions. It doesn't return a value through RETURN (uses OUT/IN OUT parameters instead).

Procedure Structure

PROCEDURE [schema.]name [(parameters)]
IS|AS
  [declarations]
BEGIN
  -- PL/SQL statements
[EXCEPTION
  exception_handlers]
END [name];

Parameter Modes

  • IN: Passes value TO procedure (default) - read-only
  • OUT: Returns value FROM procedure - write-only, no default value
  • IN OUT: Both input and output - can read and write

Calling Procedures

-- Method 1: CALL statement
CALL procedure_name(params);

-- Method 2: EXEC (SQL*Plus)
EXEC procedure_name(params);

-- Method 3: Within PL/SQL block
BEGIN
  procedure_name(params);
END;

Example: Divide Numbers

CREATE OR REPLACE PROCEDURE Divide_Numbers
  (first_number IN INTEGER, 
   second_number IN INTEGER, 
   result OUT NUMBER)
IS
BEGIN
  result := first_number / second_number;
END;

Unit 12: Exception Handling

Key Concepts

An exception is a PL/SQL error raised during execution. Exceptions can be handled to prevent program crashes.

Exception Types

  • Predefined: Oracle-defined (NO_DATA_FOUND, TOO_MANY_ROWS, ZERO_DIVIDE, etc.)
  • Non-predefined: Oracle errors without names (use PRAGMA EXCEPTION_INIT)
  • User-defined: Custom exceptions declared in DECLARE section

Exception Handling Syntax

EXCEPTION
  WHEN exception_name THEN
    -- handle exception
  WHEN OTHERS THEN
    -- handle all other exceptions
    error_code := SQLCODE;
    error_msg := SQLERRM;
END;

Key Functions

  • SQLCODE: Returns numeric error code
  • SQLERRM: Returns error message text
  • RAISE_APPLICATION_ERROR(code, message): Custom errors (-20001 to -20999)

User-Defined Exception Example

DECLARE 
  e_invalid_dept EXCEPTION;
  v_deptno NUMBER := 500;
BEGIN
  UPDATE departments SET department_name = 'Test'
  WHERE department_id = v_deptno;
  
  IF SQL%NOTFOUND THEN
    RAISE e_invalid_dept;
  END IF;
  
  COMMIT;
EXCEPTION
  WHEN e_invalid_dept THEN
    DBMS_OUTPUT.PUT_LINE('Invalid department ID');
END;

Unit 13: Packages

Key Concepts

A package groups related PL/SQL types, variables, constants, exceptions, cursors, and subprograms into a single schema object.

Package Components

  • Specification (Spec): Interface - declares public items
  • Body: Implementation - defines spec items + private items

Package Specification

CREATE OR REPLACE PACKAGE package_name IS|AS
  -- Public declarations
  variable_name datatype;
  PROCEDURE proc_name(params);
  FUNCTION func_name(params) RETURN datatype;
END package_name;

Package Body

CREATE OR REPLACE PACKAGE BODY package_name IS|AS
  -- Private declarations (optional)
  
  -- Procedure implementations
  PROCEDURE proc_name(params) IS
  BEGIN
    -- code
  END;
  
  -- Function implementations
  FUNCTION func_name(params) RETURN datatype IS
  BEGIN
    -- code
    RETURN value;
  END;
END package_name;

Advantages

  • Modularity: Group related constructs
  • Information hiding: Public vs private
  • Performance: Loaded once into memory
  • Overloading: Multiple subprograms with same name

Calling Package Elements

-- Call package procedure
package_name.procedure_name(params);

-- Call package function
v_result := package_name.function_name(params);

-- Access package variable
package_name.variable_name := value;

💡 Essential Coding Patterns

Master these patterns that appear in every lab and exam!

1. SELECT INTO - Retrieving Data

Purpose: Retrieve data from database into PL/SQL variables

Basic Pattern

DECLARE
  v_fname VARCHAR2(20);
  v_lname VARCHAR2(20);
BEGIN
  SELECT first_name, last_name 
  INTO v_fname, v_lname
  FROM employees
  WHERE employee_id = 100;
  
  DBMS_OUTPUT.PUT_LINE(v_fname || ' ' || v_lname);
END;

With Exception Handling

DECLARE
  v_salary NUMBER;
BEGIN
  SELECT salary INTO v_salary
  FROM employees WHERE employee_id = 999;
EXCEPTION
  WHEN NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE('Employee not found');
  WHEN TOO_MANY_ROWS THEN
    DBMS_OUTPUT.PUT_LINE('Multiple rows returned');
END;
⚠️ Important: SELECT INTO must return exactly ONE row, otherwise:
  • 0 rows → NO_DATA_FOUND exception
  • 2+ rows → TOO_MANY_ROWS exception

2. %TYPE and %ROWTYPE - Data Type Anchoring

%TYPE - Single Column

Purpose: Match a variable's datatype to a table column

DECLARE
  v_emp_id employees.employee_id%TYPE;    -- Same as NUMBER
  v_name employees.first_name%TYPE;        -- Same as VARCHAR2(20)
BEGIN
  v_emp_id := 100;
  SELECT first_name INTO v_name
  FROM employees WHERE employee_id = v_emp_id;
END;

%ROWTYPE - Entire Row

Purpose: Create a record structure matching all columns in a table

DECLARE
  emp_rec employees%ROWTYPE;  -- Contains all employee columns
BEGIN
  SELECT * INTO emp_rec
  FROM employees WHERE employee_id = 100;
  
  DBMS_OUTPUT.PUT_LINE('Name: ' || emp_rec.first_name);
  DBMS_OUTPUT.PUT_LINE('Salary: ' || emp_rec.salary);
  DBMS_OUTPUT.PUT_LINE('Hire Date: ' || emp_rec.hire_date);
END;

Comparison Table

Feature %TYPE %ROWTYPE
Purpose Single column datatype Complete row structure
Example v_sal employees.salary%TYPE emp_rec employees%ROWTYPE
Access v_sal := 5000; emp_rec.salary := 5000;
Use with SELECT col INTO v_sal SELECT * INTO emp_rec

3. Explicit Cursors - Processing Multiple Rows

Purpose: Process query results one row at a time (when SELECT INTO would return TOO_MANY_ROWS)

Complete Cursor Lifecycle

DECLARE
  CURSOR emp_cursor IS
    SELECT employee_id, first_name, salary
    FROM employees
    WHERE department_id = 50;
  
  v_emp_id employees.employee_id%TYPE;
  v_name employees.first_name%TYPE;
  v_salary employees.salary%TYPE;
BEGIN
  OPEN emp_cursor;                    -- 1. Open the cursor
  
  LOOP
    FETCH emp_cursor                  -- 2. Fetch one row
    INTO v_emp_id, v_name, v_salary;
    
    EXIT WHEN emp_cursor%NOTFOUND;    -- 3. Exit when no more rows
    
    DBMS_OUTPUT.PUT_LINE(v_emp_id || ' - ' || v_name || ' - ' || v_salary);
  END LOOP;
  
  CLOSE emp_cursor;                   -- 4. Close the cursor
END;

Cursor FOR Loop (Easier)

DECLARE
  CURSOR emp_cursor IS
    SELECT employee_id, first_name, salary
    FROM employees WHERE department_id = 50;
BEGIN
  FOR emp_rec IN emp_cursor LOOP
    -- No OPEN, FETCH, CLOSE needed!
    -- emp_rec is automatically created
    DBMS_OUTPUT.PUT_LINE(emp_rec.employee_id || ' - ' || 
                         emp_rec.first_name || ' - ' || 
                         emp_rec.salary);
  END LOOP;
END;

Cursor with Parameter

DECLARE
  CURSOR dept_cursor(p_dept_id NUMBER) IS
    SELECT * FROM employees WHERE department_id = p_dept_id;
BEGIN
  FOR emp IN dept_cursor(50) LOOP
    DBMS_OUTPUT.PUT_LINE(emp.first_name);
  END LOOP;
END;

4. SQL Cursor Attributes - DML Feedback

Purpose: Check results of INSERT, UPDATE, DELETE, SELECT INTO

Attribute Returns When to Use
SQL%FOUND TRUE if ≥1 row affected Check if operation succeeded
SQL%NOTFOUND TRUE if 0 rows affected Detect failed UPDATE/DELETE
SQL%ROWCOUNT Number of rows affected Get exact count

Example: UPDATE with Validation

DECLARE
  v_dept_id NUMBER := 500;
  e_invalid_dept EXCEPTION;
BEGIN
  UPDATE departments
  SET department_name = 'Testing'
  WHERE department_id = v_dept_id;
  
  IF SQL%NOTFOUND THEN               -- No rows updated
    RAISE e_invalid_dept;
  ELSE
    DBMS_OUTPUT.PUT_LINE(SQL%ROWCOUNT || ' row(s) updated');
    COMMIT;
  END IF;
EXCEPTION
  WHEN e_invalid_dept THEN
    DBMS_OUTPUT.PUT_LINE('Invalid department ID');
END;

Example: DELETE with Count

BEGIN
  DELETE FROM employees WHERE salary < 2000;
  
  DBMS_OUTPUT.PUT_LINE('Deleted ' || SQL%ROWCOUNT || ' employees');
  
  IF SQL%ROWCOUNT > 10 THEN
    ROLLBACK;
    DBMS_OUTPUT.PUT_LINE('Too many deletes - rolled back');
  ELSE
    COMMIT;
  END IF;
END;

5. PRAGMA EXCEPTION_INIT - Non-Predefined Exceptions

Purpose: Give a name to Oracle errors that don't have predefined names

Standard Pattern

DECLARE
  e_child_exists EXCEPTION;
  PRAGMA EXCEPTION_INIT(e_child_exists, -2292);  -- Foreign key violation
BEGIN
  DELETE FROM departments WHERE department_id = 50;
  COMMIT;
EXCEPTION
  WHEN e_child_exists THEN
    DBMS_OUTPUT.PUT_LINE('Cannot delete - employees exist in this dept');
END;

Common Oracle Error Codes

Error Code Meaning Example Usage
-1 Unique constraint Duplicate primary key
-1400 NOT NULL violation Required column is NULL
-2292 Foreign key violation Child records exist

Complete Example

DECLARE
  e_null_not_allowed EXCEPTION;
  PRAGMA EXCEPTION_INIT(e_null_not_allowed, -01400);
BEGIN
  INSERT INTO departments (department_id, department_name)
  VALUES (280, NULL);
EXCEPTION
  WHEN e_null_not_allowed THEN
    DBMS_OUTPUT.PUT_LINE('Department name cannot be NULL');
  WHEN OTHERS THEN
    DBMS_OUTPUT.PUT_LINE('Error: ' || SQLCODE || ' - ' || SQLERRM);
END;

6. Substitution Variables - User Input

Purpose: Get input from user at runtime (SQL*Plus/SQL Developer)

Basic Pattern

DECLARE
  v_emp_id NUMBER := &employee_id;     -- User enters value
  v_name VARCHAR2(50);
BEGIN
  SELECT first_name || ' ' || last_name INTO v_name
  FROM employees WHERE employee_id = v_emp_id;
  
  DBMS_OUTPUT.PUT_LINE('Employee: ' || v_name);
END;

With Default Prompt

ACCEPT emp_id NUMBER PROMPT 'Enter Employee ID: '

DECLARE
  v_name VARCHAR2(50);
BEGIN
  SELECT first_name INTO v_name
  FROM employees WHERE employee_id = &emp_id;
  
  DBMS_OUTPUT.PUT_LINE(v_name);
END;

7. DBMS_OUTPUT - Displaying Results

Purpose: Print messages to console (like System.out.println)

Essential Commands

-- Enable output (run once per session)
SET SERVEROUTPUT ON;

-- Basic output
DBMS_OUTPUT.PUT_LINE('Hello World');

-- Variables
DBMS_OUTPUT.PUT_LINE('Salary: ' || v_salary);

-- Concatenation
DBMS_OUTPUT.PUT_LINE(v_fname || ' ' || v_lname || ' earns ' || v_salary);

8. Complete Real-World Example

This example combines ALL the patterns above:

SET SERVEROUTPUT ON;

-- Function using cursor and %TYPE
CREATE OR REPLACE FUNCTION Get_Dept_Employee_Count(p_dept_id NUMBER)
RETURN NUMBER
IS
  v_count NUMBER := 0;
  CURSOR emp_cursor IS
    SELECT employee_id FROM employees WHERE department_id = p_dept_id;
BEGIN
  FOR emp_rec IN emp_cursor LOOP
    v_count := v_count + 1;
  END LOOP;
  RETURN v_count;
END;
/

-- Procedure using all patterns
CREATE OR REPLACE PROCEDURE Process_Department(p_dept_id IN NUMBER)
IS
  dept_rec departments%ROWTYPE;           -- %ROWTYPE
  v_emp_count NUMBER;
  e_invalid_dept EXCEPTION;
  e_no_nulls EXCEPTION;
  PRAGMA EXCEPTION_INIT(e_no_nulls, -1400);  -- Non-predefined exception
BEGIN
  -- SELECT INTO
  SELECT * INTO dept_rec 
  FROM departments WHERE department_id = p_dept_id;
  
  -- Call function
  v_emp_count := Get_Dept_Employee_Count(p_dept_id);
  
  -- SQL Cursor Attributes
  UPDATE departments 
  SET department_name = UPPER(department_name)
  WHERE department_id = p_dept_id;
  
  IF SQL%NOTFOUND THEN                    -- SQL%NOTFOUND
    RAISE e_invalid_dept;
  END IF;
  
  -- DBMS_OUTPUT
  DBMS_OUTPUT.PUT_LINE('Department: ' || dept_rec.department_name);
  DBMS_OUTPUT.PUT_LINE('Employees: ' || v_emp_count);
  DBMS_OUTPUT.PUT_LINE('Rows updated: ' || SQL%ROWCOUNT);  -- SQL%ROWCOUNT
  
  COMMIT;
EXCEPTION
  WHEN NO_DATA_FOUND THEN                -- Predefined exception
    DBMS_OUTPUT.PUT_LINE('Department ' || p_dept_id || ' not found');
  WHEN e_invalid_dept THEN               -- User-defined exception
    DBMS_OUTPUT.PUT_LINE('Invalid department ID');
  WHEN e_no_nulls THEN                   -- Non-predefined exception
    DBMS_OUTPUT.PUT_LINE('NULL values not allowed');
  WHEN OTHERS THEN
    DBMS_OUTPUT.PUT_LINE('Error: ' || SQLCODE || ' - ' || SQLERRM);
    ROLLBACK;
END;
/

-- Call with substitution variable
BEGIN
  Process_Department(&dept_id);
END;
/
